      var matchCode = new Array();

      function getXmlInstance()
      {
        var retval=null;
        try
        {
          retval=new ActiveXObject("Msxml2.XMLHTTP");
        }
        catch(e)
        {
          try
          {
            retval=new ActiveXObject("Microsoft.XMLHTTP");
          }
          catch(oc)
          {
            retval=null;
          }
        }
        if(!retval && typeof XMLHttpRequest != "undefined")
        {
          retval=new XMLHttpRequest();
        }
        return retval;
      }
      function GetMatchScore(parUrl)
      {
        var the_object;
        var UFOhttp_request = getXmlInstance();
        UFOhttp_request.open( "GET", parUrl, true );
        UFOhttp_request.onreadystatechange = function () {
        if ( UFOhttp_request.readyState == 4 ) {

        if ( UFOhttp_request.status == 200 )
        {
        
          arrayValues = UFOhttp_request.responseText;

          if(arrayValues != "")
          {
            SetArrays();
          }
        }
        
        UFOhttp_request = null;
      }
      };
      UFOhttp_request.send(null);
      }
      function SetArrays()
      {
        for(i = 0 ; i <  arrayValues.split('||').length;i++)
        {
        if(i == 0)
          addItemsToArray(Lig,arrayValues.split('||')[i]);
        if(i == 1)
          addItemsToArray(hmt,arrayValues.split('||')[i]);
        if(i == 2)
          addItemsToArray(awt,arrayValues.split('||')[i]);
        if(i == 3)
          addItemsToArray(hms,arrayValues.split('||')[i]);
        if(i == 4)
          addItemsToArray(aws,arrayValues.split('||')[i]);
        if(i == 5)
          addItemsToArray(matchSt,arrayValues.split('||')[i]);
	if(i == 6)
          addItemsToArray(matchCode,arrayValues.split('||')[i]);
        }
        if(Lig.length > 0)
        {
          document.getElementById("GreenBar").style.display= "block";
          ChangeMatchScore();
        }
      
      }

      function ChangeMatchScore()

          document.getElementById("FStatu").innerHTML = matchSt[currentMatchIndex];
	  document.getElementById("FSMatchCode").innerHTML = matchCode[currentMatchIndex];
          
          document.getElementById("SecondHolder").style.display = "none";
              $('#FirstHolder').fadeIn('slow',function () {
                setTimeout('ChangeMatchScore()',parrefrehtime);
              });
          
          
        }

      }
      function addItemsToArray(parArray,parItems)
      {
        for(k = 0 ; k <  parItems.split('##').length;k++)
        {
          parArray.push(parItems.split('##')[k]);
        }
      }
